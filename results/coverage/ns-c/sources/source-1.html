21  * Вывести его в выходной файл с именем outputName, выровняв по левому краю,
22  * чтобы длина каждой строки не превосходила lineLength.
23  * Слова в слишком длинных строках следует переносить на следующую строку.
24  * Слишком короткие строки следует дополнять словами из следующей строки.
25  * Пустые строки во входном файле обозначают конец абзаца,
26  * их следует сохранить и в выходном файле
27  */
28 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
29     val writer = File(outputName).bufferedWriter()
30     var currentLineLength = 0
31     fun append(word: String) {
32         if (currentLineLength > 0) {
33             if (word.length + currentLineLength >= lineLength) {
34                 writer.newLine()
35                 currentLineLength = 0
36             } else {
37                 writer.write(" ")
38                 currentLineLength++
39             }
40         }
41         writer.write(word)
42         currentLineLength += word.length
43     }
44     for (line in File(inputName).readLines()) {
45         if (line.isEmpty()) {
46             writer.newLine()
47             if (currentLineLength > 0) {
48                 writer.newLine()
49                 currentLineLength = 0
50             }
51             continue
52         }
53         for (word in line.split(Regex("\\s+"))) {
54             append(word)
55         }
56     }
57     writer.close()
58 }
