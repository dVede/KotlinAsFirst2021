39 fun biRoots(a: Double, b: Double, c: Double): List<Double> {
40     if (a == 0.0) {
41         return if (b == 0.0) listOf()
42         else sqRoots(-c / b)
43     }
44     val d = discriminant(a, b, c)
45     if (d < 0.0) return listOf()
46     if (d == 0.0) return sqRoots(-b / (2 * a))
47     val y1 = (-b + sqrt(d)) / (2 * a)
48     val y2 = (-b - sqrt(d)) / (2 * a)
49     return sqRoots(y1) + sqRoots(y2)
50 }
