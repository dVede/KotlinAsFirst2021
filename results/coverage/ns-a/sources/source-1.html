300 fun findSumOfTwo(list: List<Int>, number: Int): Pair<Int, Int> {
301     var result = Pair(-1, -1)

303     for (i in list.indices) {
304         val rest = number - list[i]
305         if (midRes.containsKey(list[i])) {

308         } else midRes[rest] = i
